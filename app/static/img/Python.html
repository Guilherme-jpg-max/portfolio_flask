1# Flask REST API Example
2from flask import Flask, jsonify, request
3app = Flask(__name__)
4tasks = [
5 {"id": 1, "task": "Learn Python", "done": False},
6 {"id": 2, "task": "Build Flask App", "done": False}
7]
8
9@app.route('/tasks', methods=['GET'])
10def get_tasks():
11 return jsonify({"tasks": tasks})
12
13@app.route('/tasks', methods=['POST'])
14def create_task():
15 data = request.get_json()
16 new_task = {"id": len(tasks) + 1, "task": data["task"], "done": False}
17 tasks.append(new_task)
18 return jsonify(new_task), 201
19
20@app.route('/tasks/<int:id>', methods=['PUT'])
21def update_task(id):
22 task = next((t for t in tasks if t["id"] == id), None)
23 if task:
24 task["task"] = request.json.get("task", task["task"])
25 task["done"] = request.json.get("done", task["done"])
26 return jsonify(task)
27 return jsonify({"message": "Task not found"}), 404
28
29@app.route('/tasks/<int:id>', methods=['DELETE'])
30def delete_task(id):
31 global tasks
32 tasks = [t for t in tasks if t["id"] != id]
33 return jsonify({"message": "Task deleted"})
